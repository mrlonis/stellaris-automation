planet_terraforming_flagger = {
	if = {
		limit = {
			root = {
				has_technology = tech_ecological_adaptation
			}
		}
		set_planet_flag = $planet_flag$
	}
	else_if = {
		limit = {
			is_colony = no
		}
		set_planet_flag = $planet_flag$
	}
}
